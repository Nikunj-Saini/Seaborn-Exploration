1 Basic Swarm Plot¶

In [2]:
import seaborn as sns
import matplotlib.pyplot as plt 

tips = sns.load_dataset("tips")

plt.figure(figsize=(8,6))
sns.swarmplot(x="day", y="total_bill", data=tips)
plt.title("Swarm Plot of Total Bill by Day", fontsize=16)
plt.show()
