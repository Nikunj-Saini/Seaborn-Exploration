import seaborn as sns
import matplotlib.pyplot as plt

tips = sns.load_dataset("tips")

1. Basic histogram with displot

In[2]:
sns.displot(data=tips, x="total_bill")
plt.suptitle("Displot - Basic Histogram")
plt.show()

2. Histogram with hue

In[3]:
sns.displot(data=tips, x="total_bill", hue="sex", multiple="stack")
plt.suptitle("Displot - Histogram with Hue (Stacked)")
plt.show()

3. Histogram with KDE

In[4]:
sns.displot(data=tips, x="total_bill", kde=True, bins=20)
plt.suptitle("Displot - Histogram with KDE")
plt.show()

4. FacetGrid with displot

In[5]:
sns.displot(data=tips, x="total_bill", col="time", row="smoker", kde=True)
plt.suptitle("Displot with FacetGrid")
plt.show()
