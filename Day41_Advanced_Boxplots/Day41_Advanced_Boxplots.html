import seaborn as sns
import matplotlib.pyplot as plt 

tips = sns.load_dataset("tips")
plt.figure(figsize= (8,5))
sns.boxplot(x = "total_bill", y = "day", data = tips, palette = "Set2")
plt.title("Horizontal Boxplot of Total Bill by Day")
plt.show()
