plt.figure(figsize=(10,5))
avg_scores = df.groupby("parental level of education")[["math score","reading score","writing score"]].mean().reset_index()
avg_scores = avg_scores.melt(id_vars="parental level of education", var_name="Subject", value_name="Average Score")
sns.barplot(data=avg_scores, x="parental level of education", y="Average Score", hue="Subject")
plt.xticks(rotation=45)
plt.title("Average Scores by Parental Education")
plt.show()
